import java.util.Scanner;

public class Main {
    public static void main(String[] args) {
        Scanner sc = new Scanner (System.in);
        
        int n = sc.nextInt(); 
        
        for(int i=1;i<=n;++i){ //i++ is also same because we are printing i directly
            
            System.out.print(i+" ");
            
        }
          for(int i=n;i>=1;--i){// i-- is also same because we are printing i directly
            
            System.out.print(i+" ");
          }  
    }
}
