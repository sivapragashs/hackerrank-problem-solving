import java.util.Scanner;

public class Solution {

    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        double beforetax = 0.0;

        for (int i = 0; i < 3; i++) {
            int cost = sc.nextInt();
            int qnt = sc.nextInt();
            beforetax += cost * qnt;
        }

        int budget = sc.nextInt();

        double aftertax = beforetax * 1.10;

        System.out.printf("Total (Before Tax): %.2f\n", beforetax);
        
        
        System.out.printf("Total (After Tax): %.2f\n", aftertax);
        
        
        System.out.println(aftertax <= budget ? "Within Budget" : "Over Budget");
        
        
        
       // if (aftertax <= budget) {

        //}
        
        
    }
}
